## Airflow

```bash
# DAG manuell triggern
docker exec rakuten_airflow_scheduler airflow dags trigger rakuten_incremental_training

# Scheduler/Webserver neustarten (nach DAG-Änderungen)
docker restart rakuten_airflow_scheduler rakuten_airflow_web

# Alle Tasks eines Runs löschen
docker exec rakuten_airflow_scheduler airflow tasks clear rakuten_incremental_training -y

# Live-Log eines Tasks
docker exec rakuten_airflow_scheduler bash -c \
  'RUN=$(ls -t /opt/airflow/logs/dag_id=rakuten_incremental_training/ | head -1); \
   tail -f "/opt/airflow/logs/dag_id=rakuten_incremental_training/$RUN/task_id=train_run_1/attempt=1.log"'

# Letzten Log ausgeben
docker exec rakuten_airflow_scheduler bash -c \
  'RUN=$(ls -t /opt/airflow/logs/dag_id=rakuten_incremental_training/ | head -1); \
   cat "/opt/airflow/logs/dag_id=rakuten_incremental_training/$RUN/task_id=train_run_1/attempt=1.log"' | tail -30
```

## MLflow

```bash
# Letzten Run in DB abfragen (SQLite)
sqlite3 mlflow/mlflow.db "
SELECT r.run_uuid, m.key, m.value
FROM runs r JOIN metrics m ON r.run_uuid = m.run_uuid
ORDER BY r.start_time DESC LIMIT 10;"

# Modell registrieren
python -c "
import mlflow
mlflow.set_tracking_uri('http://localhost:5000')
client = mlflow.MlflowClient()
runs = client.search_runs('1', order_by=['start_time DESC'], max_results=1)
run_id = runs[0].info.run_id
mlflow.register_model(f'runs:/{run_id}/model', 'rakuten-ice-dual-encoder')
"
```

listene best version:

docker compose exec api python -c "
from mlflow.tracking import MlflowClient
client = MlflowClient('http://mlflow:5000')
versions = client.search_model_versions(\"name='rakuten-ice-dual-encoder'\")
print(f'{\"v\":>4} {\"f1_micro\":>10} {\"f1_macro\":>10} {\"data_frac\":>10} {\"epochs\":>7} {\"run_id\":>10}')
print('-' * 65)
for v in sorted(versions, key=lambda x: int(x.version)):
    try:
        run = client.get_run(v.run_id)
        m = run.data.metrics
        p = run.data.params
        f1_mic = m.get('best_val_f1_micro', m.get('val_f1_micro', 0))
        f1_mac = m.get('best_val_f1_macro', m.get('val_f1_macro', 0))
        frac = p.get('data_fraction', '?')
        ep = p.get('epochs', '?')
        aliases = ','.join(v.aliases) if v.aliases else ''
        star = ' <- CHAMPION' if 'champion' in v.aliases else ''
        print(f'{v.version:>4} {f1_mic:>10.4f} {f1_mac:>10.4f} {frac:>10} {ep:>7}  {v.run_id[:8]}{star}')
    except Exception as e:
        print(f'{v.version:>4}  ERROR: {e}')
"
