python -c "
import mlflow
mlflow.set_tracking_uri('http://localhost:5000')
client = mlflow.MlflowClient()
runs = client.search_runs('1', order_by=['start_time DESC'], max_results=1)
run_id = runs[0].info.run_id
print('Run ID:', run_id)
model_uri = f'runs:/{run_id}/model'
mlflow.register_model(model_uri, 'rakuten-ice-dual-encoder')
"

# sq little
sqlite3 mlflow/mlflow.db "
SELECT r.run_uuid, r.start_time, p.key, p.value 
FROM runs r 
JOIN params p ON r.run_uuid = p.run_uuid 
WHERE p.key = 'max_epochs'
ORDER BY r.start_time DESC LIMIT 5;
"
